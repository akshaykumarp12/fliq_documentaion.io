void sendGift(BuildContext context, gift) {
  final message = GiftMessage(
    sentAt: DateTime.now(),
    createdAt: DateTime.now(),
    data: Gift(
      giftCatalogId: int.parse(gift.id!),
      image: gift.product!.imageUrl ?? '',
      text: '',
      coin: '${gift.coinValue}',
      seenRequestAccepted: false,
    ),
    id: '${DateTime.now().millisecondsSinceEpoch}',
    chatMessageTypeId: MessageType.gift.id,
    receiverId: param.chatPartnerId,
    senderId: context.userId,
  );
  context.read<MessagesBloc>().add(
        SendMessage(
          message: message,
          userDetails: context.read<ProfileBloc>().state.profileDetails!,
          balanceModel: context.read<GetBalanceBloc>().state.balanceModel!,
        ),
      );
}
