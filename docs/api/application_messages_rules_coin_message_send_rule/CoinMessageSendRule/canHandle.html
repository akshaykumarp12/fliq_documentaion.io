@override

/// Determines if this rule applies to the given message send event.
///
/// Returns `true` if the message is a coin type and the user’s coin balance
/// is less than the amount to send, indicating insufficient balance.
/// Otherwise, returns `false`.
bool canHandle({
  required BalanceModel balanceModel,
  required MessagesState messagesState,
  required SendMessage event,
  CustomerChatProfile? profile,
}) {
  // Check if the message is a coin message
  if (event.message.msgType == MessageType.coin) {
    // Check if the user has enough balance
    if ((balanceModel.balance?.coinBalance ?? 0) <
        (event.message as CoinMessage).data.coin) {
      return true;
    }
  }
  return false;
}
