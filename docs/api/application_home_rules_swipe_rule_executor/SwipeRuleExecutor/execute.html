Future<bool> execute({
  required BuildContext context,
  required ProfileSwiped event,
  required int totalSwipes,
  ProfileState? profileState,
  BalanceModel? balanceModel,
}) async {
  for (final guard in guards) {
    if (!await guard.canProceed()) {
      await guard.onBlocked();
      return false;
    }
  }
  for (final rule in rules) {
    final canhandle = rule.canHandle(
      event: event,
      totalSwipes: totalSwipes,
      profileState: profileState,
      balanceModel: balanceModel,
    );
    Logger.logSuccess('$canhandle, Rule: ${rule.runtimeType}');
    if (canhandle) {
      await rule.handle(
        // ignore: use_build_context_synchronously
        context: context,
        event: event,
        profileState: profileState,
        balanceModel: balanceModel,
      );
      return false;
    }
  }
  return true;
}
