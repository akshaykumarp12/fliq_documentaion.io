@override
MatchingProfile read(BinaryReader reader) {
  final numOfFields = reader.readByte();
  final fields = <int, dynamic>{
    for (int i = 0; i < numOfFields; i++) reader.readByte(): reader.read(),
  };
  return MatchingProfile(
    type: fields[0] as String?,
    id: fields[1] as String?,
    name: fields[2] as String?,
    username: fields[3] as String?,
    isActive: fields[4] as bool?,
    currentLocation: fields[5] as String?,
    square100ProfilePhotoUrl: fields[6] as String?,
    square300ProfilePhotoUrl: fields[7] as String?,
    square500ProfilePhotoUrl: fields[8] as String?,
    age: fields[9] as int?,
    avatars: (fields[10] as List).cast<Avatar>(),
    isAccountVerified: fields[11] as bool?,
    isPremiumCustomer: fields[12] as bool?,
    lastActiveAt: fields[13] as DateTime?,
    isSetActiveTheme: fields[14] as int?,
  );
}
