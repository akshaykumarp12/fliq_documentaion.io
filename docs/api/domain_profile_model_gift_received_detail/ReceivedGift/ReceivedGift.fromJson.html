factory ReceivedGift.fromJson(Map<String, dynamic> json) {
  return ReceivedGift(
    type: json["type"],
    id: json["id"],
    orderCode: json["order_code"],
    senderId: json["sender_id"],
    receiverId: json["receiver_id"],
    giftCatalogId: json["gift_catalog_id"],
    giftName: json["gift_name"],
    coins: json["coins"],
    canRevealGift: json["can_reveal_gift"],
    giftRevealRequestedAt:
        DateTime.tryParse(json["gift_reveal_requested_at"] ?? ""),
    giftRevealAcceptedAt:
        DateTime.tryParse(json["gift_reveal_accepted_at"] ?? ""),
    deliveryStatusId: json["delivery_status_id"],
    recipientName: json["recipient_name"],
    recipientPhone: json["recipient_phone"],
    recipientEmail: json["recipient_email"],
    recipientAddress: json["recipient_address"],
    recipientCountryId: json["recipient_country_id"],
    recipientStateId: json["recipient_state_id"],
    recipientCityId: json["recipient_city_id"],
    recipientCityName: json["recipient_city_name"],
    recipientZip: json["recipient_zip"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    isChatInitiated: json["is_chat_initiated"],
    isStatusReturned: json["is_status_returned"],
    sender: json["sender"] == null ? null : Receiver.fromJson(json["sender"]),
    receiver:
        json["receiver"] == null ? null : Receiver.fromJson(json["receiver"]),
  );
}
