List<Map<String, dynamic>> aboutMeAll(LoginState loginState) {
  final genders = loginState.registrationTraits.data.gender.where(
    (g) => g.name.toLowerCase() == gender?.toLowerCase(),
  );
  final genderIcon = genders.firstOrNull?.icon;

  final status = loginState.registrationTraits.data.relationshipStatus.where(
    (g) => g.name.toLowerCase() == relationshipStatusName?.toLowerCase(),
  );
  final relationshipStatus = status.firstOrNull?.icon;

  final smokingIcon = loginState.registrationTraits.data.smokingHabit.where(
    (g) => g.name == smokingHabitName,
  );
  final data = [
    {
      'image': AppAssets.height,
      'title': '${heightInCm.toString()} CM',
      'isNetworkImage': false
    },
    {
      'image': AppAssets.lang,
      'title': nativeLanguageName,
      'isNetworkImage': false
    },
    {
      'image': AppAssets.education2,
      'title': educationalQualificationName,
      'isNetworkImage': false
    },
    {
      'image': genderIcon,
      // 'image': AppAssets.women,
      'title': gender.toString(),
      'isNetworkImage': true
    },
    {
      'image': relationshipStatus,
      // 'image': AppAssets.women,
      'title': relationshipStatusName,
      'isNetworkImage': true
    },
    {
      // 'image': AppAssets.smoking2,
      'image': smokingIcon.firstOrNull?.icon,
      'title': smokingHabitName,
      'isNetworkImage': true
    },
    {
      'image': AppAssets.religion,
      'title': religionName,
      'isNetworkImage': false
    },
  ];

  return data
      .where(
        (item) =>
            item['title'] != null &&
            item['title'].toString().trim().isNotEmpty,
      )
      .toList();
}
