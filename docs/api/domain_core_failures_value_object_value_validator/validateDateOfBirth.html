Either<ValueFailure<String>, String> validateDateOfBirth(String dob) {
  if (dob.isEmpty) {
    return left(
        const ValueFailure.invalidValue('Select Date or Enter a valid date'));
  }

  // Split the date string into year, month, and day
  final parts = dob.split('-');
  if (parts.length != 3) {
    return left(const ValueFailure.invalidValue('Invalid date format'));
  }

  final year = int.tryParse(parts[0]);
  final month = int.tryParse(parts[1]);
  final day = int.tryParse(parts[2]);

  // Validate year, month, and day
  if (year == null ||
      month == null ||
      day == null ||
      day > 31 ||
      month > 12 ||
      day < 1 ||
      month < 1) {
    return left(const ValueFailure.invalidValue('Invalid date'));
  }

  // if (month < 1 || month > 12) {
  //   return left(const ValueFailure.invalidValue('Invalid month'));
  // }

  // if (day < 1 || day > 31) {
  //   return left(const ValueFailure.invalidValue('Invalid day'));
  // }

  // Validate specific month-day combinations (e.g., February 30 is invalid)
  try {
    final birthDate = DateTime(year, month, day);
    if (birthDate.year != year ||
        birthDate.month != month ||
        birthDate.day != day) {
      return left(const ValueFailure.invalidValue('Invalid date'));
    }
  } catch (e) {
    return left(const ValueFailure.invalidValue('Invalid date'));
  }

  // Calculate age
  final today = DateTime.now();
  final birthDate = DateTime(year, month, day);
  final ageDifference = today.difference(birthDate);
  final int ageInYears = (ageDifference.inDays / 365.25).floor();

  if (ageInYears < 18 || day > 31 || month > 12 || day < 1 || month < 1) {
    return left(
        const ValueFailure.invalidValue('Must be at least 18 years old'));
  }

  if (birthDate.isAfter(today)) {
    return left(const ValueFailure.invalidValue('Invalid date'));
  }

  return right(dob);
}
