ResultFuture<SocialAccountDetails> getSocialAccountDetails({
  Params? params,
  required SocialAccountType accountType,
}) async {
  final strategy = getSocialAccountStrategies[accountType];
  if (strategy == null) {
    return left(
      const MainFailure.genericError(
        errorMsg: 'Strategy not selected',
      ),
    );
  }
  getSocialAccountDetailsContext.setStrategy = strategy;

  return getSocialAccountDetailsContext.getAccountDetails(params);
}
