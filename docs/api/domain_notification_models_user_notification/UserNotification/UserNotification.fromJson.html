factory UserNotification.fromJson(Map<String, dynamic> json){
    return UserNotification(
        type: json["type"],
        id: json["id"],
        receiverId: json["receiver_id"],
        notificationCategoryId: json["notification_category_id"],
        adminNotificationId: json["admin_notification_id"],
        title: json["title"],
        description: json["description"],
        triggeredBy: json["triggered_by"],
        createdBy: json["created_by"],
        updatedBy: json["updated_by"],
        createdAt: DateTime.tryParse(json["created_at"] ?? ""),
        updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
        isCustomNotification: json["is_custom_notification"],
        notificationCategory: json["notificationCategory"] == null ? null : NotificationCategory.fromJson(json["notificationCategory"]),
        adminNotification: json["adminNotification"] == null ? null : AdminNotification.fromJson(json["adminNotification"]),
        customers: json["customers"] == null ? null : Customers.fromJson(json["customers"]),
    );
}
