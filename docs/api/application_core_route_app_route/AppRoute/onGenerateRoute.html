static Route<dynamic> onGenerateRoute(RouteSettings settings) {
  Logger.logWarning(
    'Route Name: ${settings.name},  Arguments: ${settings.arguments}, ArgumentType: ${settings.arguments.runtimeType}',
  );
  final arguments = (settings.arguments as Map<String, dynamic>?);

  switch (settings.name) {
    case HtmlViewer.routeName:
      return pushRoute(
        settings,
        HtmlViewer(
          content: arguments?['content'] ?? '',
          title: arguments?['title'] ?? '',
        ),
      );
    case ConnectionErrorScreen.routeName:
      return pushRoute(settings, ConnectionErrorScreen());
    case CreateAppealRequestScreen.routeName:
      return pushRoute(settings, CreateAppealRequestScreen());
    case AccountBlockedScreen.routeName:
      return pushRoute(settings, const AccountBlockedScreen());
    case BlockContactPopupScreen.routeName:
      return pushRoute(settings, const BlockContactPopupScreen());
    case TicketListScreen.routeName:
      return pushRoute(settings, const TicketListScreen());
    case ProfileStalkers.routeName:
      return pushRoute(settings, const ProfileStalkers());
    case VerifiedAccountScreen.routeName:
      return pushRoute(settings, const VerifiedAccountScreen());
    case CameraHomeScreen.routeName:
      List<CameraDescription> cameras = arguments!['cameras'];
      return pushRoute(
        settings,
        CameraHomeScreen(
          cameras: cameras,
        ),
      );
    case AddOccupationScreen.routeName:
      return pushRoute(settings, const AddOccupationScreen());
    case GiftAddressScreen.routeName:
      return pushRoute(settings, const GiftAddressScreen());
    case DailyLimitReached.routeName:
      return pushRoute(settings, const DailyLimitReached());
    case MatchedProfileScreen.routeName:
      String partnerProfileUrl = arguments?['partnerProfileUrl'];
      String partnerName = arguments?['partnerName'];
      int? partnerId = arguments?['partnerId'];
      return pushRoute(
          settings,
          MatchedProfileScreen(
            partnerProfileUrl: partnerProfileUrl,
            partnerName: partnerName,
            partnerId: partnerId,
          ));
    case ProfileReportScreen.routeName:
      return pushRoute(
        settings,
        ProfileReportScreen(),
      );
    case ImageSendScreen.routeName:
      final param = arguments!['param'] as MessageParterParam;
      return pushRoute(
        settings,
        ImageSendScreen(
          messageParterParam: param,
        ),
      );
    case GiftTrackingScreen.routeName:
      final orderId = arguments!['orderId'] as String;
      final isAccepted = arguments['giftRevealRequestAccepted'] ?? false;
      return pushRoute(
        settings,
        GiftTrackingScreen(
          orderId: orderId,
          giftRevealRequestAccepted: isAccepted,
          giftName: arguments['giftName'] as String,
          trackingId: arguments['trackingId'] as String?,
          orderCode: arguments['orderCode'] as String,
          userCityName: arguments['userCityName'] as String,
          createdAt: arguments['createdAt'] as DateTime?,
          updatedAt: arguments['updatedAt'] as DateTime?,
          senderId: arguments['senderId'] as int,
        ),
      );

    // case RedeemGiftScreen.routeName:
    //   return pushRoute(settings, const RedeemGiftScreen());
    // case SnapShotScreen.routeName:
    //   return pushRoute(settings, const SnapShotScreen());
    // case PreviousWinnerScreen.routeName:
    //   return pushRoute(settings, const PreviousWinnerScreen());
    // case LuckayDrawScreen.routeName:
    //   return pushRoute(settings, const LuckayDrawScreen());
    case IntarativeImageViewer.routeName:
      final image = arguments!['image'] as String;
      final isOneTimeView = arguments['isOneTimeView'] as bool?;
      final isOnVanishMode = arguments['isOnVanishMode'] as bool?;

      return pushRoute(
        settings,
        IntarativeImageViewer(
          image: image,
          isOnVanishMode: isOnVanishMode ?? false,
          isOneTimeView: isOneTimeView ?? false,
        ),
      );

    case AdvanceFilterScreen.routeName:
      return pushRoute(settings, const AdvanceFilterScreen());
    case NotificationScreen.routeName:
      return pushRoute(settings, const NotificationScreen());
    case ChatScreen.routeName:
      final param = arguments!['param'] as MessageParterParam;

      return pushRoute(
        settings,
        ChatScreen(
          param: param,
        ),
      );
    case SetYourMoodScreen.routeName:
      return pushRoute(settings, const SetYourMoodScreen());
    // case AnnonymousGiftScreen.routeName:
    //   // String partnerName = arguments?['partnerName'];
    //   return pushRoute(settings, AnnonymousGiftScreen());
    case GiftScreen.routeName:
      return pushRoute(
        settings,
        GiftScreen(
          profileId: arguments!['profileId'],
        ),
      );
    case AddReligionScreen.routeName:
      return pushRoute(settings, const AddReligionScreen());
    case AddDrinkingHabitScreen.routeName:
      return pushRoute(settings, const AddDrinkingHabitScreen());
    case AddSmokingHabitScreen.routeName:
      return pushRoute(settings, const AddSmokingHabitScreen());
    case AddGenderScreen.routeName:
      return pushRoute(settings, const AddGenderScreen());
    case AddLanguageScreen.routeName:
      return pushRoute(settings, const AddLanguageScreen());

      return pushRoute(settings, const ReferYourFriendScreen());
    case FliqPremiumScreen.routeName:
      return pushRoute(settings, const FliqPremiumScreen());

      return pushRoute(
        settings,
        PassportModeScreen(
          isProfmFilter: arguments?['isProfmFilter'],
        ),
      );
    case CompleteAccountVerificationScreen.routeName:

      final id = arguments?['id'];
      final isPremium = arguments?['isPremium'] ?? false;
      final ignorePointer = arguments?['ignorePointer'] ?? false;


      return SlideFadeRoute(
        page: SuggestedProfileDetailsScreen(
          id: id,ignorePOinter: ignorePointer,
          isPremiumCustomer: isPremium as bool,
        ),
      );

    case AppRoot.routeName:
      return pushRoute(settings, const AppRoot());
    case ProfileCreation.routeName:
      return pushRoute(settings, const ProfileCreation());
    case EmailRegistrationScreen.routeName:
      return pushRoute(settings, const EmailRegistrationScreen());
    case OtpVerificationScreen.routeName:
      return pushRoute(
        settings,
        OtpVerificationScreen(
          action: arguments!['action'] as APIAction,
        ),
      );
    case PhoneNumberRegScreen.routeName:
      return pushRoute(settings, const PhoneNumberRegScreen());
    case Login.routeName:
      return pushRoute(settings, const Login());
    case SplashScreen.routeName:
    default:
      return pushRoute(settings, const SplashScreen());
  }
