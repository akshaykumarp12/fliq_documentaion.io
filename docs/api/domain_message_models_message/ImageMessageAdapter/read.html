@override
ImageMessage read(BinaryReader reader) {
  final numOfFields = reader.readByte();
  final fields = <int, dynamic>{
    for (int i = 0; i < numOfFields; i++) reader.readByte(): reader.read(),
  };
  return ImageMessage(
    data: fields[21] as ImageData,
    sentAt: fields[9] as DateTime?,
    id: fields[2] as String?,
    receiver: fields[20] as Receiver?,
    type: fields[1] as String?,
    isOneTimeView: fields[7] as bool?,
    senderId: fields[5] as int?,
    receiverId: fields[6] as int?,
    sender: fields[14] as Sender?,
    viewedAt: fields[11] as DateTime?,
    chatMessageTypeId: fields[4] as int?,
    chatThreadId: fields[3] as int?,
    createdAt: fields[12] as DateTime?,
    deliveredAt: fields[10] as DateTime?,
    isOnVanishMode: fields[8] as bool?,
    updatedAt: fields[13] as DateTime?,
    deletedAt: fields[15] as DateTime?,
    status: fields[16] as int,
    oldMessageId: fields[17] as String?,
    messageSaved: fields[18] as bool,
    scheduledAt: fields[19] as DateTime?,
    oneTimeViewAt: fields[22] as DateTime?,
    isReadReceiptsOn: fields[23] as int?,
  );
}
