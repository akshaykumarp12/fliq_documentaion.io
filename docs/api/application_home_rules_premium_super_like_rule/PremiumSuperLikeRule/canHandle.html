@override
bool canHandle({
  required ProfileSwiped event,
  required int totalSwipes,
  ProfileState? profileState,
  BalanceModel? balanceModel,
}) {
  bool isSuperLike = event.action == ProfileSwipeAction.superLike;
  final remainingSuperLikeLimit =
      (balanceModel?.balance?.remainingSuperLikeLimit ?? 0);
  final isPremiumCustomer =
      profileState?.profileDetails?.data?.isPremiumCustomer == true;

  if (isSuperLike && isPremiumCustomer && !event.useCoins) {
    Logger.logSuccess(
      "PremiumSuperLikeRule isSuperLike: $isSuperLike,remainingSuperLikeLimit: $remainingSuperLikeLimit isPremiumCustomer: $isPremiumCustomer ",
    );

    if (remainingSuperLikeLimit <= 0 && !event.useCoins) {
      return true;
    }
    return false;
  }
  return false;
}
