Marks a message as received by updating the server and scheduling a background task.

This method:

  1. Sends a PATCH request to update the message status on the server
  2. If successful, schedules a background task to update local storage
  3. Uses Workmanager to ensure the task runs even if the app is in the background

param - Contains the message ID and user ID for the operation Returns a ResultFuture containing either a success Unit or an error

Implementation

ResultFuture<Unit> execute(
  MessageReceivedMarkParam param, {
  int retryCount = 3,
}) {
  return runSafely(
    () async {
      int attemptsLeft = retryCount;
      while (attemptsLeft > 0) {
        Logger.logWarning(
          'Attempting to set message as received: ${param.messageId}, attempts left: $attemptsLeft',
        );
        final response = await dioClient.patch(
          Urls.setMessageDelivered.replaceAll('{id}', '${param.messageId}'),
          data: {
            "data": {
              "type": "chat_messages",
              "id": num.tryParse(param.messageId ?? ''),
            }
          },
        );

        if (response.isOk) {
          return unit;
        }
        await Future.delayed(Duration(seconds: retryCount));
        attemptsLeft--;
      }

      throw CustomException(errMsg: 'Failed to set message as received');
    },
  );
}
