getDuration(int minutes) {
  int hours = minutes ~/ 60;
  int days = hours ~/ 24;
  int weeks = days ~/ 7;
  int months = days ~/ 30;
  int years = days ~/ 365;

  // Determine the display format
  String durationText;
  if (minutes < 60) {
    durationText = "$minutes MIN";
  } else if (hours < 24) {
    durationText = "$hours HR";
  } else if (days < 7) {
    durationText = "$days/${days > 1 ? "DAYS" : "DAY"}";
  } else if (weeks < 4) {
    durationText = "$weeks/${weeks > 1 ? "WEEKS" : "WEEK"}";
  } else if (months < 12) {
    durationText = "$months/${months > 1 ? "MONTHS" : "MONTH"}";
  } else {
    durationText = "$years/${years > 1 ? "YEARS" : "YEAR"}";
  }
  return durationText;
}
