String get duration {
  const int minutesInMonth = 43200; // Approx 30 days
  const int minutesInWeek = 10080;
  const int minutesInDay = 1440;
  const int minutesInHour = 60;

  int totalMinutes = this;

  int months = totalMinutes ~/ minutesInMonth;
  totalMinutes %= minutesInMonth;

  int weeks = totalMinutes ~/ minutesInWeek;
  totalMinutes %= minutesInWeek;

  int days = totalMinutes ~/ minutesInDay;
  totalMinutes %= minutesInDay;

  int hours = totalMinutes ~/ minutesInHour;
  int minutes = totalMinutes % minutesInHour;

  List<String> parts = [];

  if (months > 0) parts.add('$months ${months == 1 ? 'minute' : 'minutes'}');
  if (weeks > 0) parts.add('$weeks  ${weeks == 1 ? 'week' : 'weeks'}');
  if (days > 0) parts.add('$days ${days == 1 ? 'day' : 'days'}');
  if (hours > 0) parts.add('$hours ${hours == 1 ? 'hour' : 'hours'}');
  if (minutes > 0) {
    parts.add('$minutes ${minutes == 1 ? 'minute' : 'minutes'}');
  }
  return parts.isNotEmpty ? parts.join(' ') : '0m';
}
