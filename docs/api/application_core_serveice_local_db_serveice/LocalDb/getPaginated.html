Implementation

Future<List<T>> getPaginated<T>({
  required String localDbName,
  required int pageIndex,
  required int batchSize,
  bool Function(T)? filter,
  Comparator<T>? sortBy,
}) async {
  try {
    Logger.logError('getPaginated $localDbName: $pageIndex, $batchSize');
    final box = await _getBox<T>(localDbName);

    var allValues = box.values.where(filter ?? (_) => true).toList();
    Logger.logWarning(allValues.length);
    if (sortBy != null) {
      allValues.sort(sortBy);
    }

    final startIndex = (pageIndex - 1) * batchSize;
    if (startIndex >= allValues.length) {}

    if (startIndex > allValues.length) {
      return [];
    }

    return allValues.skip(startIndex).take(batchSize).toList();
  } on Exception catch (e) {
    Logger.logError('Error fetching data from $localDbName: $e');
    return [];
  }
}
