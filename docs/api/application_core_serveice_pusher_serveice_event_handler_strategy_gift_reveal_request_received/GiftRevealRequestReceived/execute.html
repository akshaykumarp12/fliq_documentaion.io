@override
Future<void> execute(String data) async {
  Logger.logInfo('Chat Message Id ${Japx.decode(jsonDecode(data))}');
  final messageId =
      Japx.decode(jsonDecode(data))['data']?['chatMessage']?['id'] ?? '0';
  final orderId = Japx.decode(jsonDecode(data))['data']?['id'] ?? '0';
  final name = Japx.decode(jsonDecode(data))['data']?['recipient_name'] ?? '';
  CustomDialog.showCustomDialog(
    chid: AcceptOrRejectGiftReveelRequest(
      orderId: int.parse(orderId),
      messageId: int.parse(messageId),name: name,
    ),
  );

  eventManager.emit<int>(
    eventName: MessageEvents.requestedReveelGift.name,
    data: int.parse(messageId),
  );
}
