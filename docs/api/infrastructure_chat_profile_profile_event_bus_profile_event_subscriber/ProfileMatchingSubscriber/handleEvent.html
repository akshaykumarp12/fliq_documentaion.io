void handleEvent(Event<SwipActionModel> event) async {
  final profile = CustomerChatProfile(
    chatProfile: ChatProfile(
      type: 'customers',
      isVanishModeEnabled: false,
      profileBlocked: false,
      isChatInitiated: true,
      id: event.data.partnerProfile!.id,
      messageReceivedFromPartnerAt: null,
      name: event.data.partnerProfile!.name,
      username: event.data.partnerProfile!.username,
      isActive: event.data.partnerProfile!.isActive,
      customerCode: event.data.partnerProfile!.customerCode,
      isPremiumCustomer: event.data.partnerProfile!.isPremiumCustomer,
      isOnline: event.data.partnerProfile!.isOnline,
      profilePhotoUrl: event.data.partnerProfile!.profilePhotoUrl,
      square100ProfilePhotoUrl:
          event.data.partnerProfile!.square100ProfilePhotoUrl,
      square300ProfilePhotoUrl:
          event.data.partnerProfile!.square300ProfilePhotoUrl,
      square500ProfilePhotoUrl:
          event.data.partnerProfile!.square500ProfilePhotoUrl,
      chatPartnerId: event.data.ownerId,
      chatEnabled: true,
      profileMatchedAt: DateTime.now(),
    ),
  );
  await saveChatProfile.execute([profile]);
}
