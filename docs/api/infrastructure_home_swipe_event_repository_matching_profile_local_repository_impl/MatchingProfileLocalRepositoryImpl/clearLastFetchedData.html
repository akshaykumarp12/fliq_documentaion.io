@override
ResultFuture<Unit> clearLastFetchedData() async {
  final result = await fetchMatchingProfile();

  return result.fold(
    left,
    (suggestions) async {
      for (var suggestion in suggestions) {
        await localDb.onDelete(
          localDbName: dbName,
          key: suggestion.matchingProfileId!,
        );
      }
      return right(unit);
    },
  );
}
