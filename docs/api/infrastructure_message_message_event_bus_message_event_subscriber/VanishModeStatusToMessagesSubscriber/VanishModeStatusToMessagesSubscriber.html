VanishModeStatusToMessagesSubscriber(
  super.eventManager,
  this.saveMessage,
  this.sendMessageContext,
) {
  subscribe<VanishModeLog>(
    eventName: ChatProfileEvents.updateVanishModeStatus.name,
    handler: (event) {
      saveMessage.execute([event.data]).then(
        (_) async {
          Logger.logInfo(
            'Vanish mode log saved to chat',
          );
        },
      );
      final sendMessage = sendMessageStrategies[event.data.msgType];
      if (sendMessage == null) return;
      Logger.logSuccess(
          "Message Sending VanishModeStatusToMessagesSubscriber $runtimeType, ${event.data.id}, ${sendMessage.runtimeType}");
      sendMessageContext
        ..strategy = sendMessage.strategy
        ..url = sendMessage.url;

      sendMessageContext.sendMessage(event.data);
    },
  );
}
