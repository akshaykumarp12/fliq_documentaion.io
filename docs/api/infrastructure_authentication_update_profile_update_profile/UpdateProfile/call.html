ResultFuture<UserDetailModel> call(
    BaseParams<ProfileUpdateParams> param) {
  return runSafely(() async {
    final response = await dioClient.post(
      Urls.updateProfile,
      data: param.toMap(),
    );
    Logger.logSuccess('Update customer profile param ::: ${param.toMap()}');

    if (response.isOk) {
      // final data = UserDetailModel.fromJson(response.responseData);
      // await sharedPreferences.setString(
      //   'user-details',
      //   jsonEncode(response.responseData),
      // );
      await sharedPreferences.setString(
          'user-details', jsonEncode(Japx.decode(response.data)));
      return UserDetailModel.fromJson(Japx.decode(response.data));
    }

    throw CustomException(errMsg: response.message);
  });
}
