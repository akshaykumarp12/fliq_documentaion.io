Future<void> insert<T>({
  required String localDbName,
  required T model,
  dynamic key,
}) async {
  final box = await _getBox<T>(localDbName);

  try {
    if (key != null) {
      await box.put(key, model);
    } else {
      await box.add(model);
    }
  } catch (e) {
    Logger.logError('Error insert to $localDbName: $e');
  }
}
