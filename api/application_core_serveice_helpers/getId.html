String getId(String url) {
  RegExp regex = RegExp(r'/([^/]+)\.[^.]+$');
  Match? match = regex.firstMatch(url);

  if (match != null) {
    String extractedId = match.group(1)!;
    return extractedId;
  }
  return url;
}
