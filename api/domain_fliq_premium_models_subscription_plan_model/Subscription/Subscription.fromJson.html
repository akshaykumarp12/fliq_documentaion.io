factory Subscription.fromJson(Map<String, dynamic> json) {
  return Subscription(
    type: json["type"],
    id: json["id"],
    typeId: json["type_id"],
    consoleType: json["console_type"],
    consoleProductId: json["console_product_id"],
    name: json["name"],
    description: json["description"],
    planDuration: json["plan_duration"],
    durationType: json["duration_type"],
    price: json["price"],
    offerPercentage: json["offer_percentage"],
    offerPrice: json["offer_price"],
    isActive: json["is_active"],
    isUserSwipeEnabled: json["is_user_swipe_enabled"],
    userSwipeType: json["user_swipe_type"],
    isEnabledBoostsPerWeek: json["is_enabled_boosts_per_week"],
    boostsPerWeek: json["boosts_per_week"],
    boostsPerWeekDuration: json["boosts_per_week_duration"],
    isSuperLikeEnabled: json["is_super_like_enabled"],
    superLikePerWeekLimit: json["super_like_per_week_limit"],
    messageTypeEnabled: json["message_type_enabled"],
    messageTypeLimit: json["message_type_limit"],
    isEnabledSocialMediaLinking: json["is_enabled_social_media_linking"],
    isEnabledSuperStickers: json["is_enabled_super_stickers"],
    isEnabledProfileStalkingView: json["is_enabled_profile_stalking_view"],
    isEnabledMessageBeforeMatching:
        json["is_enabled_message_before_matching"],
    isEnabledUnlimitedRewind: json["is_enabled_unlimited_rewind"],
    isEnabledUnlimitedPassport: json["is_enabled_unlimited_passport"],
    isEnabledSetMorePreferences: json["is_enabled_set_more_preferences"],
    isEnabledSeeGiftSender: json["is_enabled_see_gift_sender"],
    isEnabledPrioritizedLikes: json["is_enabled_prioritized_likes"],
    isEnabledProfileControl: json["is_enabled_profile_control"],
    isEnabledGhostMode: json["is_enabled_ghost_mode"],
    isEnabledGhostModeWithCoin: json["is_enabled_ghost_mode_with_coin"],
    createdBy: json["created_by"],
    updatedBy: json["updated_by"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    imageUrl: json["image_url"],
    featuresArray: json["features_array"] == null
        ? []
        : List<String>.from(json["features_array"]!.map((x) => x)),
  );
}
