Stream<FileUploadEvent> uploadFile(
  File file, {
  required String id,
  CancelToken? externalCancelToken,
  void Function(FileUploadEvent)? onComplete,
}) async* {
  final uploadEventController = StreamController<FileUploadEvent>.broadcast();
  final CancelToken cancelToken = externalCancelToken ?? CancelToken();

  if (cancelToken.isCancelled) {
    uploadEventController.add(
      FileUploadCanceled(FileUploadProgress(current: 0, total: 0, id: id)),
    );
    await uploadEventController.close();
    return;
  }

  Logger.logWarning('MimeType: ${lookupMimeType(file.path)}');

  Future.microtask(() async {
    try {
      uploadEventController.add(FileUploadStarted(
          FileUploadProgress(current: 0, total: 0, id: id)));

      // Prepare FormData
      final data = await _toFormData(file);
      Logger.logWarning(data.files.firstOrNull?.value.headers);

      final response = await dioClient.post(
        Urls.uploadFile,
        data: data,
        cancelToken: cancelToken,
        onSendProgress: (current, total) {
          Logger.logSuccess('Upload Progress: $current / $total');
          if (!uploadEventController.isClosed) {
            uploadEventController.add(
              FileUploading(
                FileUploadProgress(
                  id: id,
                  current: current.toDouble(),
                  total: total.toDouble(),
                ),
              ),
            );
          }
        },
      );

      if (response.isOk) {
        final uploadedFilePath =
            response.data['data']['attributes']['storage_name'];
        final progress = FileUploadProgress(
          id: id,
          current: 1,
          total: 1,
          path: uploadedFilePath,
        );
        uploadEventController.add(
          FileUploadCompleted(progress),
        );
        onComplete?.call(
          FileUploadCompleted(progress),
        );
        await Future.delayed(const Duration(milliseconds: 500));
        uploadEventController.add(
          FileUploadFinished(
            FileUploadProgress(
              id: id,
              current: 1,
              total: 1,
              path: uploadedFilePath,
            ),
          ),
        );
      } else {
        uploadEventController.add(FileUploadFailed(
          FileUploadProgress(current: 0, total: 0, id: id),
        ));
      }
    } catch (e) {
      Logger.logError('File upload error: ${e.toString()}');
      if (e is DioException && cancelToken.isCancelled) {
        uploadEventController.add(
          FileUploadCanceled(
            FileUploadProgress(current: 0, total: 0, id: id),
          ),
        );
      } else {
        uploadEventController.add(
          FileUploadFailed(
            FileUploadProgress(current: 0, total: 0, id: id),
          ),
        );
      }
    } finally {
      if (!uploadEventController.isClosed) {
        await uploadEventController.close();
      }
    }
  });

  yield* uploadEventController.stream;
}
