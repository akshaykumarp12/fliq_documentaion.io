Stream<FileDownloadEvent> downloadFile(
  String url, {
  required String fileExtention,
  CancelToken? externalCancelToken,
  void Function(FileDownloadEvent)? onComplete,
}) async* {
  final StreamController<FileDownloadEvent> downloadEventController =
      StreamController<FileDownloadEvent>.broadcast();
  final CancelToken cancelToken = externalCancelToken ?? CancelToken();

  Future.microtask(() async {
    try {
      final isGranded = await _requestPermission();

      Logger.logSuccess(dio);

      if (!isGranded) {
        downloadEventController.add(
          StoragePermissionDenied(
            FileDownloadProgress(current: 0, total: 0),
          ),
        );
      } else {
        if (path.isEmpty) {
          path = await _getPath();
        }

        downloadEventController.add(
          FileDownloadStarted(
            FileDownloadProgress(current: 0, total: 0),
          ),
        );

        final response = await dio.get(
          url,
          cancelToken: cancelToken,
          options: Options(
            responseType: ResponseType.bytes,
          ),
          onReceiveProgress: (current, total) {
            if (!downloadEventController.isClosed) {
              downloadEventController.add(
                FileDownloading(
                  FileDownloadProgress(
                    current: current.toDouble(),
                    total: total.toDouble(),
                  ),
                ),
              );
            }
          },
        );

        if (response.isOk && response.data is Uint8List) {
          final file = await writeData(
            data: response.data,
            fileName:
                'dowloaded-${DateTime.now().millisecondsSinceEpoch}.$fileExtention',
          );
          Logger.logSuccess(file.path);
          final progress = FileDownloadProgress(
            current: 1,
            total: 1,
            path: file,
            url: url,
          );
          downloadEventController.add(
            FileDownloadCompleted(progress),
          );
          onComplete?.call(FileDownloadCompleted(progress));
          await Future.delayed(const Duration(milliseconds: 500));
          downloadEventController.add(
            FileDownloadFinished(
              FileDownloadProgress(
                current: 1,
                total: 1,
                path: file,
              ),
            ),
          );
        } else {
          downloadEventController.add(
            FileDownloadFailed(
              FileDownloadProgress(current: 0, total: 0),
            ),
          );
        }
      }
    } catch (e) {
      Logger.logSuccess(e.toString());
      if (e is CustomException && e.errMsg == 'Canceld By User') {
        downloadEventController.add(
          FileDownloadCanceled(
            FileDownloadProgress(current: 0, total: 0),
          ),
        );
      } else {
        downloadEventController.add(
          FileDownloadFailed(
            FileDownloadProgress(current: 0, total: 0),
          ),
        );
      }
    } finally {
      await downloadEventController.close();
    }
  });

  yield* downloadEventController.stream;
}
