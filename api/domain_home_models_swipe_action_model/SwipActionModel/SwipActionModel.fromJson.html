factory SwipActionModel.fromJson(Map<String, dynamic> json) {
  return SwipActionModel(
    type: json["type"],
    id: json["id"],
    ownerId: json["owner_id"],
    otherUserId: json["other_user_id"],
    interactionType: json["interaction_type"],
    isLatest: json["is_latest"],
    isLikeback: json["is_likeback"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    ownerProfilePhotoUrl: json["owner_profile_photo_url"],
    ownerSquare100ProfilePhotoUrl: json["owner_square100_profile_photo_url"],
    ownerSquare300ProfilePhotoUrl: json["owner_square300_profile_photo_url"],
    ownerSquare500ProfilePhotoUrl: json["owner_square500_profile_photo_url"],
    ownerName: json["owner_name"],
    partnerProfilePhotoUrl: json["partner_profile_photo_url"],
    partnerSquare100ProfilePhotoUrl:
        json["partner_square100_profile_photo_url"],
    partnerSquare300ProfilePhotoUrl:
        json["partner_square300_profile_photo_url"],
    partnerSquare500ProfilePhotoUrl:
        json["partner_square500_profile_photo_url"],
    partnerName: json["partner_name"],
    remainingSwipeLimit: json["remaining_swipe_limit"],
    ownerProfile: json["ownerProfile"] == null
        ? null
        : OwnerProfile.fromJson(json["ownerProfile"]),
    partnerProfile: json["partnerProfile"] == null
        ? null
        : PartnerProfile.fromJson(json["partnerProfile"]),
  );
}
