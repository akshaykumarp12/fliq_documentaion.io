ResultFuture<List<Message>> execute()

Executes the fetch operation for delivered but unread messages.

This method:

  1. Makes an API call to fetch undelivered messages
  2. Transforms the response into Message objects
  3. Marks messages as saved
  4. Sorts messages by timestamp

Returns a ResultFuture containing either a list of Message or an error

Implementation

ResultFuture<List<Message>> execute() {
  return runSafely(
    () async {
      final response = await dioClient.get(
        Urls.fetchUnDeliverdMessages,
      );
      if (response.isOk) {
        log(jsonEncode(Japx.decode(response.data)));
        final messages = [
          ...(Japx.decode(response.data)['data'] as List).map(
            (message) => CustomMessageAdapter.fromMap(
              message,
            ).copyWith(
              messageSaved: true,
              deliveredAt: DateTime.now(),
              status: MessageStatus.delivered.id,
            ),
          ),
        ];
        messages.sort(_sortBy);
        // messageEvent.emitUnDeliverdMessagesDeleted(messages);
        return messages;
      }
      throw CustomException(errMsg: 'Failed to fetch user details');
    },
  );
}
