Handles user navigation triggered by notification interactions.

Routes users to corresponding app screens depending on notificationActionType. Enforces premium access for certain features. Falls back or denies navigation based on user status.

void handleNotificationNavigation({
  required BuildContext context,
  required UserNotification notification,
}) {
  final String? type = notification.adminNotification?.notificationActionType;
  Logger.logInfo(
      "notificationActionType ${notification.adminNotification?.notificationActionType}");

  switch (type) {
    case "edit-profile-screen":
      AppRoute.pushNamed(routeName: type!);
      break;
    case "daily_login_reward":
      break;

    case "coupon":
    case "coupon_notification":
      // Navigator.pushNamed(context, CouponScreen.routeName);
      break;

    case "new_users_nearby_alert":
      context.read<ThemeCubit>().setIndex(0);
      AppRoute.pushNamed(routeName: AppRoot.routeName);
      break;

    case "inventory_update":
      // Navigator.pushNamed(context, InventoryScreen.routeName);
      break;

    case "security_alert_login":
      // Navigator.pushNamed(context, LoginAlertScreen.routeName);
      break;

    case "security_alert":
      // Navigator.pushNamed(context, LoginAlertScreen.routeName);
      break;

    case "stalking_alert":
      handleFeature(
        context: context,
        onAllowed: () {
          Navigator.pushNamed(
            context,
            SuggestedProfileDetailsScreen.routeName,
            arguments: {
              'id': int.parse(
                notification.customers?.id ?? "",
              ),
              "is_premium": notification.customers?.isPremiumCustomer ?? false
            },
          );
        },
      );
      break;

    case "security_alert_change_password":
