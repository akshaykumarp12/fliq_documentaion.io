@override
Future onRequest(
    RequestOptions options, RequestInterceptorHandler handler) async {
  DateTime now = DateTime.now().toUtc();
  DeviceInfoPlugin deviceInfo = DeviceInfoPlugin();

  if (Platform.isAndroid) {
    AndroidDeviceInfo androidInfo = await deviceInfo.androidInfo;

    options.headers.addAll({
      'Device-Name': androidInfo.name,
      'Device-OS': androidInfo.version.sdkInt,
      'Device-OS-Version': androidInfo.version.release,
      'User-Agent': 'Android',
    });
  } else if (Platform.isIOS) {
    IosDeviceInfo iosInfo = await deviceInfo.iosInfo;
    options.headers.addAll({
      'Device-Name': iosInfo.name,
      'Device-OS': iosInfo.systemVersion,
      'Device-OS-Version': iosInfo.systemName,
      'User-Agent': 'iOS',
    });
  }

  options.headers.addAll({
    'Timestamp': now.millisecondsSinceEpoch,
    'Content-Type': 'application/json; charset=UTF-8',
    'Authorization':
        'Bearer ${sl<SharedPreferences>().getString('access-token') ?? ''}',
    'Access-Control-Allow-Origin': '*',
    'Accept': 'application/json',
    'Device-Type': Platform.isAndroid
        ? 1
        : Platform.isIOS
            ? 2
            : 0,
  });

  Logger.logWarning("Headers: ${options.headers.toString()}");
  Logger.logWarning("Parms: ${options.data.toString()}");
  return super.onRequest(options, handler);
}
