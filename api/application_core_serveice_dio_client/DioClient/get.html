Future<Response> get(
  String uri, {
  Map<String, dynamic>? queryParameters,
  Options? options,
  CancelToken? cancelToken,
  ProgressCallback? onReceiveProgress,
}) async {
  navigate = true;
  try {
    var response = await dio.get(
      uri,
      queryParameters: queryParameters,
      options: options,
      cancelToken: cancelToken,
      onReceiveProgress: onReceiveProgress,
    );

    return response;
  } on SocketException catch (e) {
    Logger.logError(e);
    // print("no network - 1");
    throw CustomException(
      errMsg: e.toString(),
    );
  } on FormatException catch (_) {
    throw CustomException(errMsg: "Unable to process the data");
  } catch (e) {
    Logger.logError("Dio Get Error: $e");

    if (e is DioException) {
      throw CustomException(errMsg: e.message ?? '');
    }
    if (e is DioException) {
      throw CustomException(errMsg: e.errorMessage.capitalize);
    }

    throw CustomException(errMsg: "Unknown error occurred");
  }
}
