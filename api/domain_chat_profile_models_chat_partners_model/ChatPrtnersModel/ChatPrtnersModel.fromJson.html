factory ChatPrtnersModel.fromJson(Map<String, dynamic> json) {
  return ChatPrtnersModel(
    type: json["type"],
    id: json["id"],
    authUserId: json["auth_user_id"],
    partnerId: json["partner_id"],
    isChatInitiated: json["is_chat_initiated"],
    isVanishModeEnabled: json["is_vanish_mode_enabled"],
    messageReceivedFromPartnerAt:
        DateTime.tryParse(json["message_received_from_partner_at"] ?? ""),
    typingStatus: json["typing_status"],
    typingStatusLastUpdatedAt:
        DateTime.tryParse(json["typing_status_last_updated_at"] ?? ""),
    authUser:
        json["authUser"] == null ? null : AuthUser.fromJson(json["authUser"]),
    partner:
        json["partner"] == null ? null : AuthUser.fromJson(json["partner"]),
  );
}
