ResultFuture<Unit> execute({
  required int chatPartnerId,
  required bool enable,
}) {
  return runSafely(() async {
    final url = !enable
        ? '${Urls.vanishModeBaseUrl}/$chatPartnerId${Urls.vanishModeEnable}'
        : '${Urls.vanishModeBaseUrl}/$chatPartnerId${Urls.vanishModeDisable}';

    final response = await dioClient.post(url);

    if (response.isOk) {
      final chatPartners =
          ChatPrtnersModel.fromJson(Japx.decode(response.data)['data']);

      final vanisModeChatLog = VanishModeLog(
        id: '${DateTime.now().millisecondsSinceEpoch}',
        sentAt: DateTime.now(),
        createdAt: DateTime.now(),
        senderId: chatPartners.authUserId,
        receiverId: chatPartners.partnerId,
        deliveredAt: DateTime.now(),
        viewedAt: DateTime.now(),
        chatMessageTypeId: MessageType.vanishModeLog.id,
        status: MessageStatus.seen.id,
        isOnVanishMode: chatPartners.isVanishModeEnabled,
        receiver: Receiver(
          type: '',
          id: chatPartners.partnerId.toString(),
          name: chatPartners.partner?.name,
          username: chatPartners.partner?.username,
          email: chatPartners.partner?.email,
          profilePhotoUrl: chatPartners.partner?.profilePhotoUrl,
          square100ProfilePhotoUrl:
              chatPartners.partner?.square100ProfilePhotoUrl,
          square300ProfilePhotoUrl:
              chatPartners.partner?.square300ProfilePhotoUrl,
          square500ProfilePhotoUrl:
              chatPartners.partner?.square500ProfilePhotoUrl,
        ),
        sender: Sender(
          type: '',
          id: chatPartners.authUserId.toString(),
          name: chatPartners.authUser?.name,
          username: chatPartners.authUser?.username,
          email: chatPartners.authUser?.email,
          profilePhotoUrl: chatPartners.authUser?.profilePhotoUrl,
          square100ProfilePhotoUrl:
              chatPartners.authUser?.square100ProfilePhotoUrl,
          square300ProfilePhotoUrl:
              chatPartners.authUser?.square300ProfilePhotoUrl,
          square500ProfilePhotoUrl:
              chatPartners.authUser?.square500ProfilePhotoUrl,
        ),
      );

      // Emit the vanish mode log event

      await repository.updateVanishModeStatus(
        profileId: '$chatPartnerId',
        isEnabled: !enable,
        log: vanisModeChatLog,
      );
      return unit;
    }

    throw CustomException(
      errMsg: "Failed to ${enable ? 'enable' : 'disable'} vanish mode.",
    );
  });
}
