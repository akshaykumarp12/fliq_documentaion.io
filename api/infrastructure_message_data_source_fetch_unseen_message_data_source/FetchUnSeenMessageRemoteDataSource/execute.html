@override
ResultFuture<List<Message>> execute(FetchChatParam param) {
  return runSafely(
    () async {
      final response = await dioClient.get(
        '${Urls.fetchUnSeenMessagesBetweenUsers}/${param.senderId}/${param.receiverId}',
      );
      if (response.isOk) {
        final messages = [
          ...(response.responseData as List).map(
            (message) => CustomMessageAdapter.fromMap(
              {
                'data': message,
                'included': response.data['included'],
              },
            ),
          ),
        ];
        messages.sort(_sortBy);

        return messages;
      }
      throw CustomException(errMsg: 'Failed to fetch user details');
    },
  );
}
