@override
Future<void> handle({
  required BuildContext context,
  required MessagesState messagesState,
  required BalanceModel balanceModel,
  required SendMessage event,
}) async {
  final haveEnoughCoin = (balanceModel.balance?.coinBalance ?? 0) >=
      (event.adminSettings?.coinsReqiuredForFlashMessage ?? 0);

  if (haveEnoughCoin) {
    CustomDialog.showCustomDialog(
      context: context,
      mainAxisAlignment: MainAxisAlignment.center,
      chid: UseCoinForFlashMessage(
        message: event.message,
        flashMessageLimt: event.falshMessageDailyLimt,
      ),
    );
    return;
  }

  CustomDialog.showCustomDialog(
    context: context,
    mainAxisAlignment: MainAxisAlignment.center,
    chid: const OutOfCoinPopupWidget(),
  );
}
