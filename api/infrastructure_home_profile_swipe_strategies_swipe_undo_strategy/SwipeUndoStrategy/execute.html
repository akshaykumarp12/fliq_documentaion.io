@override
ResultFuture<SwipActionModel> execute({
  required String endPoint,
  required BaseParams<SwipeProfileParam> param,
}) {
  assert(param.data.profileId != 0);
  return runSafely(
    () async {
      final response = await dioClient.delete(
          '${Urls.undoSwipe}/${param.data.profileId}/$endPoint',
          data: {
            "data": {
              "type": "profile_interaction",
              "id": param.data.profileId,
            }
          });
      if (response.isOk) {
        final swipeAction = SwipActionModel.fromJson(
          Japx.decode(response.data)['data'],
        );
        await matchingProfileLocalRepository.updateMatchingProfile(
          suggestionId: param.data.profileId,
          action: param.data.swipeAction,
          swipeAction: swipeAction,
        );
        return swipeAction;
      }
      Logger.logError(response.error);
      throw CustomException(errMsg: 'Somthing went wrong');
    },
  );
}
