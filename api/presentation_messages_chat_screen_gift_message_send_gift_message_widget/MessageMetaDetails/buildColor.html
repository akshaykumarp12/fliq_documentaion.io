Color buildColor({
  required MessageStatus status,
  required AppThemeMode theme,
}) {
  if (status == MessageStatus.seen && isReadReceiptsOn) {
    return ColorResources.oceanBlue;
  }
  switch (theme) {
    case AppThemeMode.light:
      return Colors.black;
    case AppThemeMode.dark:
    case AppThemeMode.premium:
      return Colors.white;
  }
}
