ResultFuture<Unit> acceptMessageRequest({
  required int partnerId,
}) {
  return runSafely(() async {
    final response = await dioClient.post(
      Urls.enableChat,
      data: {
        "data": {
          "type": "chat_messages",
          "attributes": {
            "partner_id": partnerId,
          }
        }
      },
    );

    if (response.isOk) {
      await messageRequestStatusHandler.execute(
        profileId: '$partnerId',
        value: true,
      );
      return unit;
    }

    throw CustomException(errMsg: response.message);
  });
}
