CurrentChatProfileSubscriberToMessage(super.eventManager, this.repository) {
  subscribe<int?>(
    eventName: MessageEvents.chatProfileChanged.name,
    handler: (event) async {
      repository.updateChatProfileId(event.data);
      if (event.data == null) {
        await FlutterWindowManagerPlus.clearFlags(
          FlutterWindowManagerPlus.FLAG_SECURE,
        );
      }
    },
  );
}
