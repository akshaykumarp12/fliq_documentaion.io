factory Data.fromJson(Map<String, dynamic> json) {
  List<T> parseList<T>(
      dynamic value, T Function(Map<String, dynamic>) fromJson) {
    if (value == null || value is! List) return [];
    return List<T>.from(
        value.map((item) => fromJson(item as Map<String, dynamic>)));
  }

  return Data(
    drinkingHabit: parseList(json['drinking_habit'], DropdownItem.fromJson),
    educationalQualification:
        parseList(json['educational_qualification'], DropdownItem.fromJson),
    lifestyleActivity:
        parseList(json['lifestyle_activity'], DropdownItem.fromJson),
    occupation: parseList(json['occupation'], DropdownItem.fromJson),
    partnerExpectations:
        parseList(json['partner_expectations'], DropdownItem.fromJson),
    personalityTrait:
        parseList(json['personality_trait'], DropdownItem.fromJson),
    relationshipStatus:
        parseList(json['relationship_status'], DropdownItem.fromJson),
    religion: parseList(json['religion'], DropdownItem.fromJson),
    sexualOrientation:
        parseList(json['sexual_orientation'], DropdownItem.fromJson),
    smokingHabit: parseList(json['smoking_habit'], DropdownItem.fromJson),
    nativeLanguage: parseList(json['native_language'], DropdownItem.fromJson),
    gender: parseList(json['gender'], Gender.fromJson),
    preferredGender: parseList(json['preferred_gender'], Gender.fromJson),
  );
}
