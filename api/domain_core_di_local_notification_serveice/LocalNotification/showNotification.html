static Future<void> showNotification({
  required String title,
  required String body,
  String? imageUrl,
  String? groupId,
}) async {
  // BigPictureStyleInformation? image;
  Uint8List? largeIcon;
  if (imageUrl != null) {
    final http.Response response = await http.get(Uri.parse(imageUrl));
    if (response.statusCode != 200) {
      throw Exception('Failed to load image');
    }

    // Convert the image to Uint8List
    // largeIcon = response.bodyBytes;
    img.Image? image = img.decodeImage(response.bodyBytes);

    // Crop into circle
    if (image != null) {
      int size = image.width < image.height ? image.width : image.height;
      image = img.copyResize(image, width: size, height: size);
      image = img.copyCrop(image, x: 0, y: 0, width: size, height: size);
      img.Image circleImage = img.copyCropCircle(image, radius: size ~/ 2);

      // Encode back to Uint8List
      largeIcon = Uint8List.fromList(img.encodePng(circleImage));
    }

    // Define the large icon and big picture
    // image = BigPictureStyleInformation(
    //   ByteArrayAndroidBitmap(largeIcon),
    //   contentTitle: title,
    //   summaryText: body,
    //   hideExpandedLargeIcon: true,
    // );
  }
  AndroidNotificationDetails androidNotificationDetails =
      AndroidNotificationDetails(
    'simple-notification',
    'Simple Notification',
    importance: Importance.max,
    // styleInformation: image,
    largeIcon: largeIcon != null ? ByteArrayAndroidBitmap(largeIcon) : null,

    priority: Priority.high,
    ticker: 'ticker',
    groupKey: groupId,
  );

  NotificationDetails notificationDetails =
      NotificationDetails(android: androidNotificationDetails);

  try {
    await flutterLocalNotificationsPlugin.show(
      _notificationId++,
      title,
      body,
      notificationDetails,
    );
  } catch (e) {
    // Handle notification display error
    Logger.logError('Error showing notification: $e');
  }
}
